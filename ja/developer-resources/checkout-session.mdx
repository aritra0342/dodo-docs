import DodoPayments from 'dodopayments';

// Initialize the Dodo Payments client
const client = new DodoPayments({
  bearerToken: process.env.DODO_PAYMENTS_API_KEY,
  environment: 'test_mode', // defaults to 'live_mode'
});

async function createCheckoutSession() {
  try {
    const session = await client.checkoutSessions.create({
      // Products to sell - use IDs from your Dodo Payments dashboard
      product_cart: [
        {
          product_id: 'prod_123', // Replace with your actual product ID
          quantity: 1
        }
      ],
      
      // Pre-fill customer information to reduce friction
      customer: {
        email: 'customer@example.com',
        name: 'John Doe',
        phone_number: '+1234567890'
      },
      
      // Billing address for tax calculation and compliance
      billing_address: {
        street: '123 Main St',
        city: 'San Francisco',
        state: 'CA',
        country: 'US', // Required: ISO 3166-1 alpha-2 country code
        zipcode: '94102'
      },
      
      // Where to redirect after successful payment
      return_url: 'https://yoursite.com/checkout/success',
      
      // Custom data for your internal tracking
      metadata: {
        order_id: 'order_123',
        source: 'web_app'
      }
    });

    // Redirect your customer to this URL to complete payment
    console.log('Checkout URL:', session.checkout_url);
    console.log('Session ID:', session.session_id);
    
    return session;
    
  } catch (error) {
    console.error('Failed to create checkout session:', error);
    throw error;
  }
}

// Example usage in an Express.js route
app.post('/create-checkout', async (req, res) => {
  try {
    const session = await createCheckoutSession();
    res.json({ checkout_url: session.checkout_url });
  } catch (error) {
    res.status(500).json({ error: 'Failed to create checkout session' });
  }
});

### 必須フィールド

<ParamField body="product_cart" type="array" required>
チェックアウトセッションに含める製品の配列。各製品にはDodo Paymentsダッシュボードから有効な`product_id`が必要です。

<Tip>
**混合チェックアウト**: 一回限りの支払い製品とサブスクリプション製品を同じチェックアウトセッションに組み合わせることができます。これにより、サブスクリプションに初期手数料を追加したり、ハードウェアバンドルとSaaSを組み合わせたり、その他の強力なユースケースが可能になります。
</Tip>

<ParamField body="return_url" type="string">
支払い完了後、顧客をリダイレクトするURL。Dodo Paymentsはリダイレクト時に以下のクエリパラメータをURLに追加します：

| パラメーター | タイプ | 条件 |
|-----------|------|-----------|
| `payment_id` | `string` | 一回限りの支払いでは常に含まれます |
| `subscription_id` | `string` | サブスクリプション支払いでは常に含まれます |
| `status` | `string` | 常に含まれます |
| `license_key` | `string` | 製品にライセンスキーが有効な場合に含まれます。複数ある場合はコンマ区切り |
| `email` | `string` | 顧客にメールが登録されている場合に含まれます |

**例: リダイレクトURL:**

```text
# One-time payment with license key
https://yoursite.com/return?payment_id=pay_xxx&status=succeeded&license_key=LK-001&email=customer%40example.com

# Subscription payment with multiple license keys
https://yoursite.com/return?subscription_id=sub_xxx&status=active&license_key=LK-001,LK-002&email=customer%40example.com

# Payment without license keys
https://yoursite.com/return?payment_id=pay_xxx&status=succeeded&email=customer%40example.com
```

<Tip>
`license_key`および`email`パラメータを使用すれば、ライセンスキーを表示したり、追加のAPIコールなしで戻り先ページで確認を送信したりできます。
</Tip>
</ParamField>

<ParamField body="minimal_address" type="boolean" default="false">
最小限の住所収集モードを有効にします。有効にするとチェックアウトでは以下のみを収集します:
- **国**: 税の判定には常に必要
- **ZIP/郵便番号**: 売上税、VAT、GSTの計算が必要な地域でのみ必要


これにより不要なフォームフィールドがなくなり、チェックアウトの摩擦が大幅に軽減されます。

<Tip>
最小限の住所を有効にすることで、チェックアウト完了をより迅速にできます。完全な請求先情報が必要な企業向けには、引き続き完全な住所収集が利用可能です。

### 11. 即時チェックアウトのための既存支払い方法の利用

顧客の保存済み支払い方法を使用して支払い方法の収集を省略し、即座に処理を行うチェックアウトセッションを作成します：

```javascript expandable
const session = await client.checkoutSessions.create({
  product_cart: [
    {
      product_id: 'prod_premium_plan',
      quantity: 1
    }
  ],
  customer: {
    customer_id: 'cus_123'  // Required when using payment_method_id
  },
  payment_method_id: 'pm_abc123',  // Use customer's saved payment method
  confirm: true,  // Required when using payment_method_id
  return_url: 'https://yourapp.com/success'
});
```

<Warning>
`payment_method_id` を使用する場合、`confirm` を `true` に設定し、既存の `customer_id` を提供する必要があります。支払い方法は支払い通貨との適合性を検証されます。
</Warning>

<Info>
支払い方法は顧客に属し、支払い通貨と互換性がなければなりません。これによりリピーター向けのワンクリック購入が可能になります。
</Info>

## Dynamic LinksからCheckout Sessionsへの移行

### 主な違い
以前、Dynamic Linksで支払いリンクを作成する際には顧客の完全な請求先住所を提供する必要がありました。

Checkout Sessionsではこれは不要になりました。持っている情報だけを渡せば、残りは当社が処理します。例えば：

- 顧客の請求先の国しか分からない場合は、その国だけを提供してください。
- チェックアウトフローは支払いページに移動する前に不足している情報を自動的に収集します。
- 一方で必要な情報をすべて持っていて支払いページに直接進ませたい場合は、完全なデータセットを渡し、リクエストボディに`confirm=true`を含めてください。
